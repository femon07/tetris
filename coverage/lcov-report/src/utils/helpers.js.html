/**
 * 2次元配列をディープコピーする
 * @param {Array} matrix - コピーする2次元配列
 * @return {Array} ディープコピーされた新しい配列
 */
export const deepCopyMatrix = (matrix) => {
  return matrix.map(row => [...row]);
};
 
/**
 * 2次元配列を転置する
 * @param {Array} matrix - 転置する2次元配列
 * @return {Array} 転置された新しい配列
 */
export const transposeMatrix = (matrix) => {
  return matrix[0].map((_, i) => matrix.map(row => row[i]));
};
 
/**
 * 行列を時計回りに90度回転させる
 * @param {Array} matrix - 回転させる2次元配列
 * @return {Array} 回転された新しい配列
 */
export const rotateMatrix = (matrix) => {
  const n = matrix.length;
  const result = Array(n).fill().map(() => Array(n).fill(0));
  
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      result[j][n - 1 - i] = matrix[i][j];
    }
  }
  
  return result;
};
 
/**
 * 2つの位置が等しいかどうかを判定する
 * @param {Object} pos1 - 位置1 {x, y}
 * @param {Object} pos2 - 位置2 {x, y}
 * @return {boolean} 等しければtrue
 */
export const arePositionsEqual = (pos1, pos2) => {
  return pos1.x === pos2.x && pos1.y === pos2.y;
};
 
/**
 * ランダムな整数を生成する
 * @param {number} min - 最小値（含む）
 * @param {number} max - 最大値（含む）
 * @return {number} 生成されたランダムな整数
 */
export const getRandomInt = (min, max) => {
  return Math.floor(Math.random() * (max - min + 1)) + min;
};
 
/**
 * ミリ秒を待機する
 * @param {number} ms - 待機するミリ秒
 * @return {Promise} 指定時間後に解決するPromise
 */
export const sleep = (ms) => {
  return new Promise(resolve => setTimeout(resolve, ms));
};
 
/**
 * スコアを計算する
 * @param {number} linesCleared - 消したライン数 (0-4)
 * @param {number} currentScore - 現在のスコア
 * @return {number} 新しいスコア
 */
export const calculateScore = (linesCleared, currentScore = 0) => {
  if (linesCleared === 0) return currentScore;
  
  // テトリス公式のスコア計算に基づく
  const linePoints = [0, 40, 100, 300, 1200];
  const level = Math.floor(currentScore / 1000) + 1;
  
  return currentScore + (linePoints[linesCleared] || 0) * level;
};
 
/**
 * レベルを計算する
 * @param {number} lines - 消したラインの合計
 * @param {number} linesPerLevel - レベルアップに必要なライン数
 * @return {number} 現在のレベル
 */
export const calculateLevel = (lines, linesPerLevel = 10) => {
  return Math.floor(lines / linesPerLevel) + 1;
};
 
/**
 * ゲームスピードを計算する
 * @param {number} level - 現在のレベル
 * @param {Object} speedTable - レベルごとのスピードテーブル
 * @return {number} ミリ秒単位のスピード
 */
export const calculateGameSpeed = (level, speedTable) => {
  return speedTable[Math.min(level, Object.keys(speedTable).length)] || speedTable[Object.keys(speedTable).length];
};
 
/**
 * モバイルデバイスかどうかを判定する
 * @return {boolean} モバイルデバイスの場合はtrue
 */
export const isMobileDevice = () => {
  return /Android|webOS|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
};
 
/**
 * タッチイベントを処理する
 * @param {Event} e - タッチイベント
 * @return {Object} 正規化された位置 {x, y}
 */
export const getTouchPosition = (e) => {
  if (!e.touches) return null;
  const touch = e.touches[0] || e.changedTouches[0];
  const rect = e.target.getBoundingClientRect();
  
  return {
    x: touch.clientX - rect.left,
    y: touch.clientY - rect.top,
  };
};
